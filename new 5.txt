  public static function BMR($gender,$height,$weight,$age,$stress=1)
    {
      if($gender=="male")
      {
         $result = 66.47 + 13.75*$weight + 5*$height  - 6.76*$age*$stress;
         return $result ;
      }
      $result =    655.1 + 9.56*$weight + 1.85*$height- 4.68*$stress;
